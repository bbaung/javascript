focus management

//The focus() method in JavaScript is used to set the focus on a specified element, such as an input field or a button. When an element is focused, it becomes the active element in the document, meaning it is ready to receive user input or interaction.

var button = document.getElementById("myButton");

button.focus();

alert(document.activeElement === button); //true

//The hasFocus() method is used to determine if the document or any element within it has focus. It returns a boolean value: true if the document or any element within it has focus, and false otherwise.

var button = document.getElementById("myButton");

button.focus();

alert(document.hasFocus());
